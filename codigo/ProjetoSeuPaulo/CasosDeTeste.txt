 */

public class CadastroPlantaMain {

    /**
     * @param nomePlanta
     * @param lote
     * @param data
     * @return 
     */
    public  PlantaDados cadastrarPlanta(String nomePlanta, int lote, LocalDate data){
        if(nomePlanta == null ||nomePlanta.isBlank()){
            System.out.println("Nome da planta não pode ser vazio");
        }
        if(lote <= 0){
            System.out.println("Lote da planta não pode ser vazio");
        }
        if(data == null){
            System.out.println("Data da planta não pode ser vazio");
        }
        return new PlantaDados(nomePlanta,lote,data);
        
    }


}
